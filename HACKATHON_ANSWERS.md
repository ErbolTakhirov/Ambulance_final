```python
def _astar(self, start, goal):
    # Priority queue: (f_score, node)
    open_set = [(0, start)]
    
    # g_score: —Ä–µ–∞–ª—å–Ω–∞—è —Å—Ç–æ–∏–º–æ—Å—Ç—å –æ—Ç —Å—Ç–∞—Ä—Ç–∞ –¥–æ —É–∑–ª–∞

    # –î–ª—è –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏—è –ø—É—Ç–∏
    came_from = {}
    
    while open_set:
        current = heappop(open_set)[1]
        
        if current == goal:
            return self._reconstruct_path(came_from, current)
        
        # –ò—Å—Å–ª–µ–¥—É–µ–º —Å–æ—Å–µ–¥–µ–π
        for neighbor, edge_cost in self.graph[current]:
            # –ù–æ–≤–∞—è —Å—Ç–æ–∏–º–æ—Å—Ç—å –ø—É—Ç–∏
            tentative_g = g_score[current] + edge_cost
            
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                came_from[neighbor] = current
                g_score[neighbor] = tentative_g
                
                # A* —ç–≤—Ä–∏—Å—Ç–∏–∫–∞: f = g + h
                # h = —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –ø–æ –ø—Ä—è–º–æ–π –¥–æ —Ü–µ–ª–∏ (Haversine)
                h = self._haversine(neighbor[0], neighbor[1], goal[0], goal[1])
                f_score = tentative_g + h
                
                heappush(open_set, (f_score, neighbor))
    
    return []  # –ü—É—Ç—å –Ω–µ –Ω–∞–π–¥–µ–Ω
```

```python
def _haversine(lat1, lng1, lat2, lng2):
    R = 6371  # –†–∞–¥–∏—É—Å –ó–µ–º–ª–∏ –≤ –∫–º
    
    dlat = radians(lat2 - lat1)
    dlng = radians(lng2 - lng1)
    
    a = sin(dlat/2)¬≤ + cos(lat1) * cos(lat2) * sin(dlng/2)¬≤
    c = 2 * asin(sqrt(a))
    
    return R * c  # –∫–º
```
